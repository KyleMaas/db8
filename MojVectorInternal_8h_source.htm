140 template<class T, class EQ, class COMP>
141 int MojVector<T, EQ, COMP>::compare(const MojVector& comp) const
142 {
143  ConstIterator iter = begin();
144  ConstIterator compIter = comp.begin();
145  for (;;) {
146  if (iter == end())
147  return compIter == comp.end() ? 0 : -1;
148  if (compIter == comp.end())
149  return 1;
150  int comp = COMP()(*iter, *compIter);
151  if (comp != 0)
152  return comp;
153  ++iter;
154  ++compIter;
155  }
156  MojAssertNotReached();
157  return 0;
158 }

265 template<class T, class EQ, class COMP>
266 MojErr MojVector<T, EQ, COMP>::erase(MojSize idx, MojSize numElems)
267 {
268  MojAssert(idx + numElems <= size());
269 
270  // may invalidate iters
271  MojErr err = ensureWritable();
272  MojErrCheck(err);
273  MojSize numMove = size() - (idx + numElems);
274  Iterator pos = m_begin + idx;
275  // do moves
276  for (MojSize i = 0; i < numMove; ++i, ++pos) {
277  *pos = *(pos + numElems);
278  }
279  // then destroy elems hanging off the end
280  m_end -= numElems;
281  MojDestroy(m_end, numElems);
282 
283  return MojErrNone;
284 }

332 template<class T, class EQ, class COMP>
333 MojErr MojVector<T, EQ, COMP>::shift(MojSize idx, MojSize numElems)
334 {
335  MojAssert(idx <= size());
336  MojSize numMove = size() - idx;
337  MojSize numCopyConstruct = MojMin(numElems, numMove);
338  MojSize numDefaultConstruct = numElems - numCopyConstruct;
339  MojSize numAssign = numMove - numCopyConstruct;
340 
341  // make space (possibly invalidating iterator)
342  MojSize err = ensureSpace(size() + numElems);
343  MojErrCheck(err);
344  Iterator elem = m_end - 1;
345  // incrementing end as we go to make sure newly created elems
346  // get destroyed if a constructor throws
347  for (MojSize i = 0; i < numDefaultConstruct; ++i, ++m_end) {
348  new(m_end) T();
349  }
350  for (MojSize i = 0; i < numCopyConstruct; ++i, ++m_end) {
351  ConstIterator src = m_end - numElems;
352  new(m_end) T(*src);
353  }
354  for (MojSize i = 0; i < numAssign; ++i, --elem) {
355  ConstIterator src = elem - numElems;
356  *elem = *src;
357  }
358 
359  return MojErrNone;
360 }

388 template<class T, class EQ, class COMP>
389 MojErr MojVector<T, EQ, COMP>::realloc(MojSize numElems)
390 {
391  // we use a temporary vector and keep it valid at all times to make
392  // sure that proper cleanup happens if a constructor throws an exception
393  MojVector<T, EQ, COMP> tmp;
394  tmp.m_begin = static_cast<T*>(MojRefCountAlloc(numElems * sizeof(T)));
395  MojAllocCheck(tmp.m_begin);
396  tmp.m_end = tmp.m_begin;
397  tmp.m_endAlloc = tmp.m_begin + numElems;
398  for (ConstIterator i = begin(); i < end(); ++i, ++tmp.m_end) {
399  new(tmp.m_end) T(*i);
400  }
401  swap(tmp);
402 
403  return MojErrNone;
404 }
