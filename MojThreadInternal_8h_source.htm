345 template <class T, class CTOR>
346 MojThreadLocalValue<T, CTOR>::~MojThreadLocalValue()
347 {
348  if (m_key != MojInvalidThreadKey) {
349  // main thread tss won't get cleanup up unless we do it here
350  T* val = static_cast<T*>(MojThreadGetSpecific(m_key));
351  MojErr err = MojThreadSetSpecific(m_key, NULL);
352  if (err == MojErrNone) {
353  delete val;
354  } else {
355  MojAssertNotReached();
356  }
357 
358  err = MojThreadKeyDelete(m_key);
359  MojErrCatchAll(err);
360  }
361 }

363 template <class T, class CTOR>
364 MojErr MojThreadLocalValue<T, CTOR>::get(T*& valOut)
365 {
366  valOut = NULL;
367  if (m_key == MojInvalidThreadKey)
368  MojErrThrow(MojErrNotInitialized);
369 
370  valOut = static_cast<T*>(MojThreadGetSpecific(m_key));
371  if (valOut == NULL) {
372  CTOR ctor;
373  MojAutoPtr<T> newVal(ctor());
374  MojAllocCheck(newVal.get());
375  MojErr err = MojThreadSetSpecific(m_key, newVal.get());
376  MojErrCheck(err);
377  valOut = newVal.release();
378  }
379  return MojErrNone;
380 }
