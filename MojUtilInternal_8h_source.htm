107 template<class T, class COMP>
108 MojSize MojBinarySearch(T key, const T* array, MojSize numElems)
109 {
110  MojAssert (array || numElems == 0);
111 
112  MojSize low = 0;
113  MojSize high = numElems;
114  while (low < high) {
115  MojSize mid = low + ((high - low) / 2);
116  int comp = COMP()(key, array[mid]);
117  if (comp == 0)
118  return mid;
119  else if (comp > 0)
120  low = mid + 1;
121  else
122  high = mid;
123  }
124  return MojInvalidSize;
125 }

133 template<class T, class COMP>
134 MojSize MojQuickSortPartition(T* array, MojSize left, MojSize right, const COMP& comp)
135 {
136  // pick random pivot
137  unsigned int seed = (unsigned int) left;
138  MojSize pivot = left + (MojRand(&seed) % (right - left));
139  MojSize last = right - 1;
140  MojSize mid = left;
141  // move pivot to end
142  MojSwap(array[pivot], array[last]);
143  // move all elements < pivot to left of mid
144  for (MojSize i = left; i < last; ++i) {
145  if (comp(array[i], array[last]) < 0) {
146  MojSwap(array[i], array[mid]);
147  ++mid;
148  }
149  }
150  // move pivot to mid
151  MojSwap(array[mid], array[last]);
152 
153  return mid;
154 }

158 {
